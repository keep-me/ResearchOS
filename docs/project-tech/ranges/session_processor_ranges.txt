2705:     def apply_event(self, event_name: str, data: dict[str, Any]) -> list[tuple[str, dict[str, Any]]]:

2788:         parsed = runtime._coerce_runtime_event(raw)
2789:         if parsed is None:
2790:             return []
2791:         event_name, data = parsed
2792:         return [runtime._format_sse_event(name, payload) for name, payload in self.apply_event(event_name, data)]

2794:     def stream(
2795:         self,
2796:         raw_stream: Iterator[Any],
2797:         *,
2798:         manage_lifecycle: bool = True,
2799:         control: Any | None = None,
2800:         lifecycle_kind: str = "prompt",
2801:         step_index: int = 0,
2802:         publish_bus: bool = False,
2803:     ) -> Iterator[str]:
2804:         stream_control = control if isinstance(control, PromptStreamControl) else None
2805:         driver = (
2806:             PromptEventStreamDriver(
2807:                 control=stream_control,
2808:                 session_id=self.session_id,
2809:                 lifecycle_kind=lifecycle_kind,
2810:                 step_index=step_index,
2811:             )
2812:             if stream_control is not None
2813:             else None
2814:         )
2815:         if manage_lifecycle:
2816:             self.start()
2817:         try:
2818:             for raw in raw_stream:
2819:                 for item in self.consume(raw):
2820:                     if driver is None:
2821:                         yield item
2822:                     else:
2823:                         yield from driver.emit_raw(item, publish_bus=publish_bus)
2824:                 parsed = runtime._coerce_runtime_event(raw)
2825:                 if parsed is None:
2826:                     serialized = str(raw)
2827:                     if driver is None:
2828:                         yield serialized
2829:                     else:
2830:                         yield from driver.emit_raw(serialized, publish_bus=publish_bus)
2831:                 else:
2832:                     event_name, data = parsed
2833:                     serialized = runtime._format_sse_event(event_name, data)
2834:                     if driver is None:
2835:                         yield serialized
2836:                     else:
2837:                         yield from driver.emit_raw(serialized, publish_bus=publish_bus)
2838:         finally:
2839:             self.finalize(
2840:                 manage_lifecycle=manage_lifecycle,
2841:                 handed_off=bool(getattr(control, "handed_off", False)),
2842:             )

2844:     def finalize(self, *, manage_lifecycle: bool = True, handed_off: bool = False) -> None:
2845:         if runtime.is_session_aborted(self.session_id) and self.state.pending_error is None and self.state.current_message_id:
2846:             self._apply_error("会话已中止")
2847:         if manage_lifecycle and not handed_off:
2848:             runtime.set_session_status(self.session_id, {"type": "idle"})
2849:         if manage_lifecycle:
2850:             runtime.clear_session_abort(self.session_id)
